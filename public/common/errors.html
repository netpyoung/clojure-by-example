(defn myDivision [x y]
  (try
    (/ x y)
    (catch ArithmeticException e
      (println "Exception message: " (.getMessage e)))
    (finally
      (println "Done."))))
 
(myDivision 4 2)
 
;; >> Done.
=> 2
 
(myDivision 4 0)
 
;; >> Exception message:  Divide by zero
>> Done.
=> nil
