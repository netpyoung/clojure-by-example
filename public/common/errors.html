(defn myDivision [x y]
  (try
    (/ x y)
    (catch ArithmeticException e
      (println &quot;Exception message: &quot; (.getMessage e)))
    (finally
      (println &quot;Done.&quot;))))
 
(myDivision 4 2)
 
;; &gt;&gt; Done.
=&gt; 2
 
(myDivision 4 0)
 
;; &gt;&gt; Exception message:  Divide by zero
&gt;&gt; Done.
=&gt; nil
