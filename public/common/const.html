(ns common.const)
<p>http://stackoverflow.com/questions/9162558/how-does-clojure-const-work</p> 
(def ^:const s &quot;constant&quot;)
 
(def ^:const n 500000000)
 
(def ^:const d (/ 3e20 n))
 
(println s)
 
;; &gt;&gt; constant
=&gt; nil
 
(println d)
 
;; &gt;&gt; 6.0E11
=&gt; nil
 
(println (long d))
 
;; &gt;&gt; 600000000000
=&gt; nil
 
(println (Math/sin n))
 
;; &gt;&gt; -0.28470407323754404
=&gt; nil
