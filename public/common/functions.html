(ns common.functions)
 
(defn plus [a b]
  (+ a b))
 
(def ret (plus 1 2))
 
(println &quot;1+2 =&quot; ret)
 
;; &gt;&gt; 1+2 = 3
=&gt; nil
 
(defn ret-vals []
  [3 7])
 
(let [[a b] (ret-vals)]
  (println a)
  (println b))
 
;; &gt;&gt; 3
&gt;&gt; 7
=&gt; nil
 
(let [[_ c] (ret-vals)]
  (println c))
 
;; &gt;&gt; 7
=&gt; nil
