(require '[clojure.core.strint :as strint])
(strint/<< "p is ~{(pr-str p)}")
;; => "p is #common.string_formatting.Point{:x 1, :y 2}"
(let [w (java.io.StringWriter.)]
  (pp/pprint p w)
  (.toString w))
;; => "{:x 1, :y 2}\n"
