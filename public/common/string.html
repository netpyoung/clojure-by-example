(ns common.string)
 
(require '[clojure.string :as s])
 
(some #{\e \s} "test")
 
;; => \e
 
(count (filter #{\t} "test"))
 
;; => 2
 
(re-find #"^te\w*" "test")
 
;; => "test"
 
(re-find #"\w*st$" "testx")
 
;; => "test"
 
(.indexOf (seq "test") \e)
 
;; => 1
