(require '[cemerick.url :as url])
 
(require '[org.bovinegenius [exploding-fish :as uri]])
 
(def s "postgres://user:pass@host.com:5432/path?k=v#f")
(def u (uri/uri s))
 
(:scheme u)
 
;; => "postgres"
 
(:user-info u)
 
;; => "user:pass"
 
(:host u)
 
;; => "host.com"
 
(:port u)
 
;; => 5432
 
(:path u)
 
;; => "/path"
 
(:fragment u)
 
;; => "f"
 
(:query u)
 
;; => "k=v"
 
(->> "k"
     (get (->> (uri/query-pairs u)
               (into {}))))
 
;; => "v"
