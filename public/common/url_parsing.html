(require '[cemerick.url :as url])
 
(require '[org.bovinegenius [exploding-fish :as uri]])
 
(def s &quot;postgres://user:pass@host.com:5432/path?k=v#f&quot;)
(def u (uri/uri s))
 
(:scheme u)
 
;; =&gt; &quot;postgres&quot;
 
(:user-info u)
 
;; =&gt; &quot;user:pass&quot;
 
(:host u)
 
;; =&gt; &quot;host.com&quot;
 
(:port u)
 
;; =&gt; 5432
 
(:path u)
 
;; =&gt; &quot;/path&quot;
 
(:fragment u)
 
;; =&gt; &quot;f&quot;
 
(:query u)
 
;; =&gt; &quot;k=v&quot;
 
(-&gt;&gt; &quot;k&quot;
     (get (-&gt;&gt; (uri/query-pairs u)
               (into {}))))
 
;; =&gt; &quot;v&quot;
