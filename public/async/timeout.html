(ns async.timeout)
<p>https://gobyexample.com/timeouts</p> 
(require '[clojure.core.async :as async :refer [&lt;! &gt;! &lt;!! timeout chan alt! go go-loop put! alts! alts!!]])
 
(def c1 (chan 1))
 
(go (&lt;! (timeout (* 2 1000)))
    (&gt;! c1 &quot;result 1&quot;))
 
(let [[v ch] (alts!! [c1 (timeout 1000)])]
  (if v
    (println v)
    (println &quot;timeout 1&quot;)))
 
;; &gt;&gt; timeout 1
 
(def c2 (chan 1))
 
(go (&lt;! (timeout (* 2 1000)))
    (&gt;! c2 &quot;result 2&quot;))
 
(let [[v ch] (alts!! [c2 (timeout 3000)])]
  (if v
    (println v)
    (println &quot;timeout 2&quot;)))
 
;; &gt;&gt; result 2
