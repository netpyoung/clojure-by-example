(require '[clojure.core.async :as async :refer [&lt;! &gt;! &lt;!! &gt;!! timeout chan alt! go go-loop put! pipe close! map&lt;]])
 
(defn worker [id jobs results]
  (go-loop []
    (when-let [j (&lt;! jobs)]
      (println &quot;worker&quot; id, &quot;processing job&quot; j)
      (&lt;! (timeout 3000))
      (&gt;! results (* j 2))
      (recur))))
