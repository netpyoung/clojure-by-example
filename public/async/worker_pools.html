(require '[clojure.core.async :as async :refer [<! >! <!! >!! timeout chan alt! go go-loop put! pipe close! map<]])
 
(defn worker [id jobs results]
  (go-loop []
    (when-let [j (<! jobs)]
      (println "worker" id, "processing job" j)
      (<! (timeout 3000))
      (>! results (* j 2))
      (recur))))
