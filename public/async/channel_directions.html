(ns async.channel-directions)
 
(require '[clojure.core.async :as async :refer [&lt;! &gt;! &lt;!! &gt;!! timeout chan alt! go go-loop put! pipe]])
 
(defn ping [pings msg]
  (put! pings msg))
 
(defn pong [pings pongs]
  (-&gt;&gt; (&lt;!! pings)
       (put! pongs)))
 
(def pi (chan 1))
(def po (chan 1))
 
(ping pi &quot;passed message&quot;)
(pong pi po)
 
(println (&lt;!! po))
 
;; &gt;&gt; passed message
=&gt; nil
 
(defn ping' [pings msg]
  (go
    (&gt;! pings msg)))
 
(defn pong' [pings pongs]
  (go
    (-&gt;&gt; (&lt;! pings)
         (&gt;! pongs))))
 
(def pi' (chan 1))
(def po' (chan 1))
 
(ping pi' &quot;passed message&quot;)

=&gt; nil
