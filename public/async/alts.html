(require '[clojure.core.async :as async :refer [&lt;! &gt;! &lt;!! timeout chan alt! go go-loop put! alts! alts!!]])
 
(def c1 (chan))
(def c2 (chan))
 
(go (&lt;! (timeout 1000))
    (&gt;! c1 &quot;one&quot;))
 
(go (&lt;! (timeout (* 2 1000)))
    (&gt;! c2 &quot;two&quot;))
 
(dotimes [i 2]
  (let [[v ch] (alts!! [c1 c2])]
    (condp = ch
      c1 (println &quot;received&quot; v)
      c2 (println &quot;received&quot; v)
      nil)))
