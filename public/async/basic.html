(require '[clojure.core.async :as async :refer [<! >! <!! timeout chan alt! go]])
 
(defn f [from]
  (dotimes [i 3]
    (println from ":" i)))
 
(f :direct)

=> nil
 
(do (go (f :goroutine))
    (go (println :going)))
